# Scanner Scorecard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Track post-signal price action for every scanner event and surface signal quality metrics (MFE, MAE, R-multiple, expectancy) for EdgeExplorer.

**Architecture:** Two new tables (`scanner_outcome_snapshots`, `scanner_outcome_summaries`) fed by a Celery periodic task that reads from existing `stock_aggregates`. A shared `DataReadinessService` validates data availability before any subsystem runs. New `/api/outcomes/` router exposes scorecard, interval, distribution, and edge-decay endpoints. The `ScannerService._save_event()` hook creates pending snapshots when events fire.

**Tech Stack:** Python 3.11, FastAPI, SQLAlchemy 2.0 (sync sessions), PostgreSQL (JSONB), Celery + Redis, Pydantic v2, pytest with SQLite

```python
# backend/app/services/data_readiness.py
"""
DataReadinessService — validates that required aggregate data exists for a
ticker before scanners, outcome capture, or analytics run.
"""

import logging
from dataclasses import dataclass, field
from datetime import date, timedelta
from typing import Dict, List, Optional

from sqlalchemy import func
from sqlalchemy.orm import Session

from app.models.scanner_config import ScannerConfig
from app.models.stock_aggregate import StockAggregate

logger = logging.getLogger(__name__)


@dataclass
class TimespanCoverage:
    timespan: str
    multiplier: int
    required_from: date
    required_to: date
    available_from: Optional[date] = None
    available_to: Optional[date] = None
    gap_ranges: List[tuple] = field(default_factory=list)
    is_ready: bool = False


@dataclass
class ReadinessReport:
    ticker: str
    scanner_type: str
    coverages: List[TimespanCoverage] = field(default_factory=list)
    is_ready: bool = True
    missing_summary: str = ""


class DataReadinessService:
    """Shared service for checking and ensuring aggregate data availability."""

    @staticmethod
    def check(
        db: Session,
        ticker: str,
        scanner_type: str,
        as_of: Optional[date] = None,
    ) -> ReadinessReport:
        """Check whether required aggregate data exists for the given ticker."""
        if as_of is None:
            as_of = date.today()

        config = (
            db.query(ScannerConfig)
            .filter(ScannerConfig.scanner_type == scanner_type)
            .first()
        )

        report = ReadinessReport(ticker=ticker, scanner_type=scanner_type)

        if not config or not config.data_requirements:
            return report

        reqs = config.data_requirements
        timespans = reqs.get("timespans", [])
        missing_parts = []

        for ts_req in timespans:
            timespan = ts_req["timespan"]
            multiplier = ts_req.get("multiplier", 1)
            lookback_days = ts_req.get("lookback_days", 30)

            required_from = as_of - timedelta(days=lookback_days)
            required_to = as_of

            row = db.query(
                func.min(func.date(StockAggregate.timestamp)).label("earliest"),
                func.max(func.date(StockAggregate.timestamp)).label("latest"),
            ).filter(
                StockAggregate.ticker == ticker,
                StockAggregate.timespan == timespan,
                StockAggregate.multiplier == multiplier,
                func.date(StockAggregate.timestamp) >= required_from,
                func.date(StockAggregate.timestamp) <= required_to,
            ).first()

            available_from = row.earliest if row else None
            available_to = row.latest if row else None
            is_ready = (available_from is not None and available_to is not None
                        and available_from <= required_from + timedelta(days=2)
                        and available_to >= required_to - timedelta(days=2))

            coverage = TimespanCoverage(
                timespan=timespan,
                multiplier=multiplier,
                required_from=required_from,
                required_to=required_to,
                available_from=available_from,
                available_to=available_to,
                is_ready=is_ready,
            )
            report.coverages.append(coverage)

            if not is_ready:
                report.is_ready = False
                if available_from is None:
                    missing_parts.append(f"{timespan} bars: no data found")
                else:
                    missing_parts.append(
                        f"{timespan} bars: have {available_from} to {available_to}, "
                        f"need {required_from} to {required_to}"
                    )

        report.missing_summary = "; ".join(missing_parts) if missing_parts else ""
        return report

    @staticmethod
    def ensure(
        db: Session,
        ticker: str,
        scanner_type: str,
        as_of: Optional[date] = None,
    ) -> bool:
        """Check readiness and fetch missing data from Polygon if needed."""
        report = DataReadinessService.check(db, ticker, scanner_type, as_of=as_of)
        if report.is_ready:
            return True

        from app.services.stock_data import StockDataService

        for coverage in report.coverages:
            if coverage.is_ready:
                continue
            try:
                days_needed = (coverage.required_to - coverage.required_from).days + 1
                StockDataService.refresh_stock_data(
                    db, ticker,
                    timespan=coverage.timespan,
                    multiplier=coverage.multiplier,
                    period=f"{days_needed}d",
                )
            except Exception as e:
                logger.error(
                    "DataReadinessService.ensure: failed to fetch %s %s bars for %s: %s",
                    coverage.timespan, coverage.multiplier, ticker, e,
                )
                return False

        final_report = DataReadinessService.check(db, ticker, scanner_type, as_of=as_of)
        return final_report.is_ready

    @staticmethod
    def check_batch(
        db: Session,
        tickers: List[str],
        scanner_type: str,
        as_of: Optional[date] = None,
    ) -> Dict[str, ReadinessReport]:
        """Batch readiness check for multiple tickers."""
        return {
            ticker: DataReadinessService.check(db, ticker, scanner_type, as_of=as_of)
            for ticker in tickers
        }
```

```python
"""
OutcomeService — creates, captures, and summarises scanner outcome data.
"""

import logging
from datetime import datetime, timezone
from decimal import Decimal
from typing import List, Optional

from sqlalchemy.orm import Session

from app.models.scanner_config import ScannerConfig
from app.models.scanner_event import ScannerEvent
from app.models.scanner_outcome_snapshot import ScannerOutcomeSnapshot
from app.models.scanner_outcome_summary import ScannerOutcomeSummary

logger = logging.getLogger(__name__)


class OutcomeService:
    """Service for managing scanner outcome tracking."""

    @staticmethod
    def create_pending_snapshots(
        db: Session,
        event: ScannerEvent,
    ) -> List[ScannerOutcomeSnapshot]:
        """Create pending snapshot rows for each configured interval."""
        config = (
            db.query(ScannerConfig)
            .filter(ScannerConfig.scanner_type == event.scanner_type)
            .first()
        )
        if not config or not config.outcome_config:
            return []

        oc = config.outcome_config
        intervals = oc.get("intervals", [])
        if not intervals:
            return []

        ref_source = oc.get("reference_price_source", "opening_price")
        reference_price = getattr(event, ref_source, None) or event.opening_price
        if reference_price is None:
            logger.warning(
                "OutcomeService: no reference price for event %s (source=%s), skipping snapshots",
                event.id, ref_source,
            )
            return []

        snapshots = []
        for interval_key in intervals:
            snapshot = ScannerOutcomeSnapshot(
                scanner_event_id=event.id,
                interval_key=interval_key,
                reference_price=reference_price,
                status="pending",
            )
            db.add(snapshot)
            snapshots.append(snapshot)

        db.flush()
        return snapshots
```

```python
    @staticmethod
    def capture_snapshot(db: Session, snapshot: ScannerOutcomeSnapshot) -> None:
        """Fill in a pending snapshot's price metrics from stock_aggregates."""
        from app.models.stock_aggregate import StockAggregate

        event = db.query(ScannerEvent).filter(
            ScannerEvent.id == snapshot.scanner_event_id
        ).first()
        if not event:
            snapshot.status = "failed"
            return

        event_date = event.event_date
        ref_price = float(snapshot.reference_price)

        # Determine the time window based on interval_key
        from datetime import timedelta, time as _time
        from zoneinfo import ZoneInfo

        _ET = ZoneInfo("America/New_York")
        day_open_et = datetime.combine(event_date, _time(9, 30), tzinfo=_ET)
        day_open_utc = day_open_et.astimezone(timezone.utc).replace(tzinfo=None)

        interval_map = {
            "1h": timedelta(hours=1),
            "4h": timedelta(hours=4),
            "eod": timedelta(hours=6, minutes=30),  # 9:30 -> 16:00
            "1d": timedelta(days=1),
            "2d": timedelta(days=2),
            "5d": timedelta(days=5),
        }
        delta = interval_map.get(snapshot.interval_key)
        if delta is None:
            snapshot.status = "failed"
            return

        window_start = day_open_utc
        window_end = day_open_utc + delta

        bars = (
            db.query(StockAggregate)
            .filter(
                StockAggregate.ticker == event.ticker,
                StockAggregate.timespan == "minute",
                StockAggregate.timestamp >= window_start,
                StockAggregate.timestamp < window_end,
            )
            .order_by(StockAggregate.timestamp.asc())
            .all()
        )

        if not bars:
            snapshot.status = "failed"
            return

        high = max(float(b.high) for b in bars)
        low = min(float(b.low) for b in bars)
        last_close = float(bars[-1].close)
        total_volume = sum(int(b.volume) for b in bars)

        snapshot.snapshot_price = Decimal(str(round(last_close, 4)))
        snapshot.pct_change = Decimal(str(round((last_close - ref_price) / ref_price * 100, 4)))
        snapshot.high_since_signal = Decimal(str(round(high, 4)))
        snapshot.low_since_signal = Decimal(str(round(low, 4)))
        snapshot.volume_since_signal = total_volume
        snapshot.captured_at = datetime.now(timezone.utc).replace(tzinfo=None)
        snapshot.status = "captured"
```

```python
    @staticmethod
    def recompute_summary(db: Session, scanner_event_id: int) -> Optional[ScannerOutcomeSummary]:
        """Upsert the outcome summary for an event from its captured snapshots."""
        event = db.query(ScannerEvent).filter(ScannerEvent.id == scanner_event_id).first()
        if not event:
            return None

        snapshots = (
            db.query(ScannerOutcomeSnapshot)
            .filter(ScannerOutcomeSnapshot.scanner_event_id == scanner_event_id)
            .all()
        )
        captured = [s for s in snapshots if s.status == "captured"]
        if not captured:
            return None

        ref_price = float(captured[0].reference_price)
        if ref_price <= 0:
            return None

        # MFE/MAE across all captured snapshots
        highest = max(float(s.high_since_signal) for s in captured if s.high_since_signal is not None)
        lowest = min(float(s.low_since_signal) for s in captured if s.low_since_signal is not None)
        mfe_pct = round((highest - ref_price) / ref_price * 100, 4)
        mae_pct = round((lowest - ref_price) / ref_price * 100, 4)
        mfe_mae_ratio = round(abs(mfe_pct / mae_pct), 4) if mae_pct != 0 else None

        # EOD snapshot for follow-through and eod_pct_change
        eod_snap = next((s for s in captured if s.interval_key == "eod"), None)
        eod_pct = float(eod_snap.pct_change) if eod_snap and eod_snap.pct_change else None

        # Follow-through check
        config = (
            db.query(ScannerConfig)
            .filter(ScannerConfig.scanner_type == event.scanner_type)
            .first()
        )
        threshold = 2.0
        if config and config.outcome_config:
            threshold = config.outcome_config.get("follow_through_threshold_pct", 2.0)
        follow_through = eod_pct is not None and eod_pct >= threshold

        # Completeness check
        all_intervals = set()
        if config and config.outcome_config:
            all_intervals = set(config.outcome_config.get("intervals", []))
        captured_intervals = {s.interval_key for s in captured}
        is_complete = all_intervals.issubset(captured_intervals) if all_intervals else len(captured) > 0

        # Upsert summary
        summary = (
            db.query(ScannerOutcomeSummary)
            .filter(ScannerOutcomeSummary.scanner_event_id == scanner_event_id)
            .first()
        )
        if not summary:
            summary = ScannerOutcomeSummary(scanner_event_id=scanner_event_id)
            db.add(summary)

        summary.reference_price = Decimal(str(ref_price))
        summary.mfe_pct = Decimal(str(mfe_pct))
        summary.mae_pct = Decimal(str(mae_pct))
        summary.mfe_mae_ratio = Decimal(str(mfe_mae_ratio)) if mfe_mae_ratio is not None else None
        summary.eod_pct_change = Decimal(str(eod_pct)) if eod_pct is not None else None
        summary.follow_through = follow_through
        summary.is_complete = is_complete
        summary.completed_at = datetime.now(timezone.utc).replace(tzinfo=None) if is_complete else None

        db.flush()
        return summary
```

```python
@celery_app.task(bind=True, max_retries=1)
def capture_outcome_snapshots(self):
    """
    Periodic task: capture pending outcome snapshots whose interval has elapsed.
    Runs every 30 min during market hours, once at 16:30 ET, and once at 20:00 ET.
    """
    from app.models.scanner_outcome_snapshot import ScannerOutcomeSnapshot
    from app.services.outcome_service import OutcomeService

    db: Session = SessionLocal()
    try:
        pending = (
            db.query(ScannerOutcomeSnapshot)
            .filter(ScannerOutcomeSnapshot.status == "pending")
            .all()
        )
        if not pending:
            return

        logger.info(f"capture_outcome_snapshots: processing {len(pending)} pending snapshots")

        captured_count = 0
        events_to_recompute = set()

        for snapshot in pending:
            try:
                OutcomeService.capture_snapshot(db, snapshot)
                if snapshot.status == "captured":
                    captured_count += 1
                    events_to_recompute.add(snapshot.scanner_event_id)
            except Exception as e:
                logger.error(
                    "capture_outcome_snapshots: failed snapshot %s: %s",
                    snapshot.id, e,
                )
                snapshot.status = "failed"

        db.commit()

        for event_id in events_to_recompute:
            try:
                OutcomeService.recompute_summary(db, event_id)
            except Exception as e:
                logger.error(
                    "capture_outcome_snapshots: failed summary recompute for event %s: %s",
                    event_id, e,
                )

        db.commit()
        logger.info(
            "capture_outcome_snapshots: captured %d/%d snapshots, recomputed %d summaries",
            captured_count, len(pending), len(events_to_recompute),
        )

    except Exception as e:
        logger.error(f"capture_outcome_snapshots failed: {e}")
        db.rollback()
        raise self.retry(exc=e, countdown=120)
    finally:
        db.close()
```

```python
        avg_mfe = sum(float(s.mfe_pct or 0) for s in complete) / n

        avg_r = None

        if r_values:
            avg_r = sum(r_values) / len(r_values)

        # Expectancy = (win_rate * avg_win) - (loss_rate * avg_loss)
        avg_win = sum(float(s.eod_pct_change or 0) for s in wins) / len(wins) if wins else 0
```

```typescript
  return response.data;
};
```
